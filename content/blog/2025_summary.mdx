export const pubs2025 = pubs.filter(pub => {
  const year = pub.issued?.["date-parts"]?.[0]?.[0]
  const isNotArxivOnly = pub["container-title"] !== "arXiv"
  return (year === 2025 || year === "2025") && isNotArxivOnly
}).sort((a, b) => {
  const yearA = a.issued?.["date-parts"]?.[0]?.[0] || 0
  const yearB = b.issued?.["date-parts"]?.[0]?.[0] || 0
  return yearB - yearA
})

export const pubs2026 = pubs.filter(pub => {

}).sort((a, b) => {
  const yearA = a.issued?.["date-parts"]?.[0]?.[0] || 0
  const yearB = b.issued?.["date-parts"]?.[0]?.[0] || 0
  return yearB - yearA
})
